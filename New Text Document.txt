create table department(
		dp_id int auto_increment,
        dp_name varchar(45),
        PRIMARY KEY (dp_id)
);
create table employee(
	em_id int auto_increment,
    em_name varchar(45),
    em_salary decimal(10,2),
    em_permanent boolean,
    em_date_of_birth date,
    em_dp_id int,
    PRIMARY KEY (em_id),
    FOREIGN KEY (em_dp_id) REFERENCES department(dp_id)
);


create table skill(
	sk_id int auto_increment,
    sk_name varchar(45),
    PRIMARY KEY (sk_id)
);

create table employee_skill(
	es_id int auto_increment,
    es_em_id int,
    es_sk_id int,
    PRIMARY KEY (es_id),
    FOREIGN KEY (es_em_id) REFERENCES employee(em_id),
    FOREIGN KEY (es_sk_id) REFERENCES skill(sk_id)
    
);

create table attempt
(
at_id int primary key auto_increment,
at_date date,
at_us_id int,
at_score double,
foreign key(at_us_id) references user(us_id)
);

create table user
(
us_id int primary key auto_increment,
us_name varchar(45),
us_email varchar(45)
);


create table question
(
